Create a list of characters [‘a’, ‘b’, ‘c’, ‘d’, ‘e’, ‘f’, ‘g’, ‘h’, ‘i’, ‘j’]. Use random.choice to randomly select a letter from this list. Then, use the index method to find the position of the selected letter in the “listified” string Your mother was a hamster and your father smells of elderberries. If the letter is found, print its index; otherwise, print a message saying the letter is not in the string.

# 1
import random

# Step 1: Create a list of characters
char_list = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j']

# Step 2: Randomly select a letter from the list
selected_letter = random.choice(char_list)
print(f"Selected letter: {selected_letter}")

# Step 3: Convert the string to a list of characters
string = "Your mother was a hamster and your father smells of elderberries."
string_list = list(string.lower())  # Convert to lowercase to match char_list

# Step 4: Find the index of the selected letter
if selected_letter in string_list:
    index = string_list.index(selected_letter)
    print(f"The letter '{selected_letter}' is found at index {index}.")
else:
    print(f"The letter '{selected_letter}' is not in the string.")

Use a while loop to reverse the backwardsText list without using the reverse() method. Print the reversed list. Compare it to doing the same reversal, but using the reverse() method

#2
# Initialize the list
backwardsText = ['Inquisition!', 'Spanish', 'the', 'expects', 'Nobody']

# Reverse the list using a while loop
reversed_list = []
index = len(backwardsText) - 1

while index >= 0:
    reversed_list.append(backwardsText[index])
    index -= 1

# Print the reversed list using the while loop
print("Reversed using while loop:", reversed_list)

# Reverse the list using reverse() method
backwardsText_copy = backwardsText[:]  # Make a copy of the list
backwardsText_copy.reverse()

# Print the reversed list using reverse() method
print("Reversed using reverse() method:", backwardsText_copy)

Create a list of 10 random integers. Use a while loop to remove all integers greater than 50 from the list. Print the modified list.

import random

# Step 1: Create a list of 10 random integers between 1 and 100
random_integers = []
for _ in range(10):
    random_integers.append(random.randint(1, 100))
print("Original list:", random_integers)

# Step 2: Use a while loop to remove all integers greater than 50
while True:
    found = False
    for num in random_integers:  # Iterate over the original list
        if num > 50:
            random_integers.remove(num)
            found = True  # Set flag to True if a number is removed
            break  # Exit the loop after removing one element
    if not found:
        break  # Exit the while loop if no numbers were removed

# Step 3: Print the modified list
print("Modified list:", random_integers)
